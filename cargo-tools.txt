# Modern CLI tools installed via cargo
# These replace traditional Unix tools with better defaults
#
# Usage: grep -v '^#\|^$' cargo-tools.txt | sed 's/#.*//' | xargs cargo install

bat          # cat with syntax highlighting
fd-find      # find replacement
ripgrep      # grep replacement
lsd          # ls replacement
git-delta    # git pager with syntax highlighting
procs        # ps replacement
zoxide       # cd replacement (smart directory jumper)
starship     # cross-shell prompt
